import java.util.Scanner;

public class multiplication {
    public static void main(String[] args) {
        System.out.println("enter a num");
        Scanner p=new Scanner(System.in);
        int n=p.nextInt();
        for(int i=10;i>=1;i--)
        {
            System.out.printf("%d*%d=%d",n,i,n*i);
            System.out.println("\n");
        }
    }
}
